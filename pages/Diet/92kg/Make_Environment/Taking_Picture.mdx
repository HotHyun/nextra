흔히들 **‘눈바디’** 라고 하죠. 자기 자신이 판단하기에가 아닌, 다른 사람이 판단하기에 체형의 변화가 있는가를 확인하기 위해서는 **‘사진’** 을 찍는 것이 중요합니다. 거울을 보면 자기 자신이 얼마나 달라졌는가를 확실히 확인하기 위해서는 오랜 시간이 걸립니다. 그러면 그러한 변화가 빠르게 일어나지 않아 다이어트를 포기하는 사람들이 많이 나오고는 합니다. 그것을 막기 위해서는 사진을 찍어놓고, 어느 변화가 있는지를 기록하는 것이 다이어트의 하나의 동기가 될 수 있습니다. 자기 자신만의 **사진 찍기 주기**를 정해두고 전신 사진을 찍어보세요! 자다가도 이불을 걷어찰 만큼 부끄러운 사진이겠지만, 자신의 상태를 **정확히 직시**해야만 더욱 **객관적인 다이어트**를 할 수 있습니다 !!

import PageKey from "../../../../components/PageKey";
import { useRouter } from "next/router";
import { useEffect } from "react";
import mixpanel from "mixpanel-browser";

export const Final = () => {
  const router = useRouter();
  useEffect(() => {
    const timer = setTimeout(() => {
      mixpanel.track("View Web Interactive Content Page", {
        contentId: "Making Environment Taking Picture",
      });
    }, 1000);
  }, []);
  return (
    <PageKey
      route={router.pathname}
      EventName="Making Environment Taking Picture"
    />
  );
};

<Final />
